// ... الكود السابق (CSS و HTML) يبقى كما هو function renderRow(v, isNew = false) { const tbody = document.getElementById('tableBody'); const row = tbody.insertRow(isNew ? 0 : -1); row.setAttribute('data-created', v.created || v.date); row.innerHTML = ` ${getDisplayNote(v.notes || "[]")} ${v.updated || '-'} `; } function insertNewRow() { const today = new Date().toISOString().split('T')[0]; // التأكد من تمرير كائن يحتوي على حقل email renderRow({ comp:'', mgr:'', mob:'', email:'', cr:'', curr:'', prop:'', status:'', date: today, notes: '[]', updated: '-', created: today }, true); toggleSaveBtn(true); } function saveAllData() { const rows = document.querySelectorAll('#tableBody tr'); const allData = []; rows.forEach(row => { const ins = row.querySelectorAll('.excel-input'); let updatedAt = row.querySelector('.updated-at').innerText; if (row.getAttribute('data-pending-update')) updatedAt = row.getAttribute('data-pending-update'); // ترتيب الـ index هنا مهم جداً ويجب أن يطابق ترتيب الـ inputs في renderRow allData.push({ comp: ins[0].value, mgr: ins[1].value, mob: ins[2].value, email: ins[3].value, // هذا هو الترتيب الصحيح الآن cr: ins[4].value, curr: ins[5].value, prop: ins[6].value, status: row.querySelector('select').value, date: ins[7].value, notes: row.querySelector('.notes-cell').getAttribute('data-notes'), updated: updatedAt, created: row.getAttribute('data-created') }); }); localStorage.setItem('asgate_visits_data', JSON.stringify(allData)); loadSavedData(); }
