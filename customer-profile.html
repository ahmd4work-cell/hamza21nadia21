function saveNewVisit() { // 1. تجهيز بيانات الزيارة/الفرصة const entryDate = document.getElementById('v-date').value; const entryStatus = document.getElementById('v-status').value; const entryNotes = document.getElementById('v-notes').value; const entryData = { comp: clientName, mgr: currentCustomer.mgr, mob: currentCustomer.mob, cr: currentCustomer.cr, curr: document.getElementById('v-curr').value, prop: document.getElementById('v-prop').value, status: entryStatus, date: entryDate, // تنسيق الملاحظات ليتوافق مع نظام الاستيكر notes: JSON.stringify([{ date: "من ملف العميل", text: entryNotes }]), updated: '-', created: new Date().toISOString().split('T')[0] }; if (entryStatus === "مهتم") { // 2. إذا كانت الحالة "مهتم" -> نرسلها إلى الفرص البيعية let allOpps = JSON.parse(localStorage.getItem('asgate_opportunities_data') || '[]'); allOpps.push(entryData); localStorage.setItem('asgate_opportunities_data', JSON.stringify(allOpps)); alert("تم تحويل العميل إلى قائمة 'الفرص البيعية' بنجاح."); } else { // 3. أي حالة أخرى -> نرسلها إلى سجل الزيارات العام let allVisits = JSON.parse(localStorage.getItem('asgate_visits_data') || '[]'); allVisits.push(entryData); localStorage.setItem('asgate_visits_data', JSON.stringify(allVisits)); alert("تم حفظ الزيارة في 'سجل الزيارات العام'."); } // 4. تحديث العرض في الصفحة الحالية (ملف العميل يعرض كل شيء يخصه) loadClientVisits(); closeVisitModal(); // تنظيف الحقول للزيارة القادمة document.getElementById('v-curr').value = ''; document.getElementById('v-prop').value = ''; document.getElementById('v-notes').value = ''; } // تعديل بسيط في دالة التحميل لتعرض الزيارات والفرص معاً في ملف العميل function loadClientVisits() { const allVisits = JSON.parse(localStorage.getItem('asgate_visits_data') || '[]'); const allOpps = JSON.parse(localStorage.getItem('asgate_opportunities_data') || '[]'); // دمج السجلين لعرض تاريخ العميل الكامل في ملفه الشخصي const fullHistory = [...allVisits, ...allOpps].filter(item => item.comp === clientName); const body = document.getElementById('visits-log-body'); if (fullHistory.length === 0) { body.innerHTML = 'لا توجد سجلات سابقة.'; return; } // ترتيب من الأحدث للأقدم fullHistory.sort((a, b) => new Date(b.date) - new Date(a.date)); body.innerHTML = ''; fullHistory.forEach(v => { let noteText = ""; try { const parsed = JSON.parse(v.notes); noteText = parsed[parsed.length-1].text; } catch(e) { noteText = v.notes || "-"; } body.innerHTML += ` ${v.date} ${v.curr || '-'} ${v.prop || '-'} ${v.status} ${noteText} `; } }
